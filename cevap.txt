public static void main(String[] args){
    int list[] = {3,8,20,12,15,25,30,9};
    for (int i = 0; i < list.length; i++) {
        if (list[i] % 3 == 0){
            if(list[i] % 5 == 0)
                System.out.println("FizzBuzz");
            else
                System.out.println("Fizz");
        } else if(list[i] % 5 == 0)
            System.out.println("Buzz");
    }
}
